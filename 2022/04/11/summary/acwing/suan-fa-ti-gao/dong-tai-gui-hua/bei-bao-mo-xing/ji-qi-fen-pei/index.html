package main

import (
	"fmt"
)

var f, a [30][40]int

func max(a, b int) int {
	if a > b {
		return a
	}
	return b
}

/*
	1013. 机器分配
	https://www.acwing.com/problem/content/1015/
	题目描述
		每个公司获得机器的数量有对应的收益
		求在给出的机器数量怎么分配求收益最大
	Tag：
		1.dp
		2.dp求具体方案
	思路：
		倒着来一遍dp ,f[i][j]：表示当前位置i之后的（n-i）个公司分配j个机器的最大盈利
		然后从前往后找解，这样找出的解也就是一条最段路必然是最小字典序
*/

//f[i][j]:前i个公司分配j个机器的最大盈利
func main() {
	var n, m int = 0, 0
	fmt.Scan(&n, &m)

	for i := 1; i <= n; i++ {
		for j := 1; j <= m; j++ {
			fmt.Scan(&a[i][j])
		}
	}
	//倒着来一个dp
	for i := n; i >= 1; i-- {
		for j := 0; j <= m; j++ { //当前可以选择的个数
			for k := 0; k <= j; k++ { //当前要选择的个数
				f[i][j] = max(f[i][j], f[i+1][j-k]+a[i][k])
			}
		}
	}
	//从前往后找最短路
	maxv := f[1][m]
	fmt.Println(maxv)
	for i := 1; i <= n; i++ {
		for j := 0; j <= m; j++ {
			if m >= j && maxv == f[i+1][m-j]+a[i][j] {
				fmt.Println(i, j)
				maxv = f[i+1][m-j]
				m -= j
				break
			}
		}
	}
}
