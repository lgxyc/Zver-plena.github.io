#include<bits/stdc++.h>
#define eps 1e-8
typedef long long ll;
typedef unsigned long long ull;
typedef std::pair<int, int> PII;
void _IOS(){ std::ios::sync_with_stdio(0);std::cin.tie();std::cout.tie();}
void _DEBUG(){freopen("input.txt","r",stdin);freopen("debug.txt","w",stdout);}
void _TEST(){freopen("input.txt","r",stdin);freopen("output.txt","w",stdout);}
void __F(){exit(0);}
using namespace std;
/*
	https://www.acwing.com/problem/content/4418/
	点的赋值
    题目描述
        给你一个无向图，每一条边的两个端点之和必须是奇数    
        端点只能是1，或2，或3
    Tag：
        1.二分图
    思路：
        通过相邻点不存在相同奇偶性来进行染色，如果不满足这样的情况答案必然是0
        反之

        通过奇偶来染色，可以通过1,2来进行染色通过 3-x 来转换奇偶
        因为对于每个奇数可以取两个值，且每一种情况可以交替一次，
        对于奇数占的位置为s1,偶数占的位置为s2
        结果为 pow(2,s1)*pow(1,s2)  (1)
        位置交替一次
        奇数占的位置为s2,偶数占的位置为s1
        结果为 pow(2,s2)*pow(1,s1)  (2)

        所以总的结果为  (1)+(2)
        

*/
const int MOD=998244353;
const int N=300010,M=2*N;
int e[M],ne[M],h[N],idx;
int st[N];
int s1,s2;
void add(int a,int b){
    ne[idx]=h[a],e[idx]=b,h[a]=idx++;
}
int  get(int k ){
    int res=1;
    while(k--) res=((ll)res*2)%MOD;
    return res;
}
bool  dfs(int u,int c){

    st[u]=c;
    if(c==1) s1++;
    else s2++;
    for(int i=h[u];i!=-1;i=ne[i]){
        int j =e[i];
        if(st[j]==0 && !dfs(j,3-c)) return false;
        else if(st[j]==c) return false;
    }
    return true;
}
void solve(){
    int n ,m;
    cin>>n>>m;
    memset(h,-1,(n+1)*4);
    memset(st,0,(n+1)*4);
    int a,b;
    for(int i=1;i<=m;i++){
        cin>>a>>b;
        add(a,b);
        add(b,a);
    }
    int res=1;
    for(int i=1;i<=n;i++){
        if(st[i]==0){
            s1=s2=0;
            if(!dfs(i,1)){
                res=0;
                break;
            }
            res = (ll)res*(get(s1)+get(s2)) %MOD;
        }
    }
    cout<<res<<endl;

}
int main()
{
    _IOS();
    int t ;
    cin>>t;
    while(t--){
        solve();
    }

    return 0;
}
