Lowest Common Ancestor

/*
 * @Autor: Zver

#include<bits/stdc++.h>
#define eps 1e-8
typedef long long ll;
typedef unsigned long long ull;
typedef std::pair<int, int> PII;
void __IOS(){ std::ios::sync_with_stdio(0);std::cin.tie();std::cout.tie();}
void __DEBUG(){freopen("input.txt","r",stdin);freopen("debug.txt","w",stdout);}
void __TEST(){freopen("input.txt","r",stdin);freopen("output.txt","w",stdout);}
void __F(){exit(0);}
using namespace std;
/*

    倍增LCA
    dep[i]:  i  的深度
    f[i][j]: i 向上跳 2^j 步到达的点
    LCA倍增过程
        1.预处理 dep[]和f[][] 在 bfs（root）函数中
          f[i][j]的状态转移
          i ----- A ------ B
             2^j-1   2^j-1
             
          i可以先走 2^j-1 步 到 A  : f[i][k-1]  = A
          A再走 2^j-1 到 B   : f[A][k-1] = B
			
          所以状态转移方程就是
          f[i][B] = f[A][j-1]
                  = f[f[i][k-1]][j-1];
                  
        2.LCA过程
            1.a 跳到和 b 同层次 
            2.a 和 b 同时向上跳
            如果f[a][k] == f[b][k]
                只能说明f[a][k]为a和b的common Ancestor
                但不能说明为 Least common Ancestor
            如果f[a][k] != f[b][k]
                说明f[a][k] 还没有到LCA 往上跳


*/
const int N = 4e4 + 10;
int dep[N],f[N][21];
int n,m,a,b;
vector<int >head[N];
void bfs(int root){
    memset(dep,0x3f,sizeof(dep));
    // 哨兵
    dep[0]=0;
    // 初始化
    dep[root]=1;

    queue<int>q;
    q.push(root);
    while(q.size()){
        int u = q.front();
        q.pop();
        for(int x: head[u]){
            // 如果当前点没有被更新
            if(dep[x] > dep[u] + 1 ){
                dep[x] = dep[u] + 1;
                q.push(x);
                f[x][0] = u;
                // dp
                for(int k=1;k<=20;k++){
                    f[x][k]=f[f[x][k-1]][k-1];
                }
            }
        }
    }
}
int lca(int a, int b){

    if(dep[a] < dep[b]) swap(a,b);
    // a跳到b同深度
    for(int k=20;k>=0;k--){
        if(dep[f[a][k]] >= dep[b]){
            a = f[a][k];
        }
    }
    if(a==b)return a;
    // a和b一起向上跳
    for(int k=20;k>=0;k--){
        // 如果不相等说明不是 common ancestor
        // 可以往上跳
        if(f[a][k] != f[b][k]){
            a = f[a][k];
            b = f[b][k];
        }
    }
    return f[a][0];
}
int main()
{
    int n ;

    cin>>n;
    for(int i=1;i<=n;i++){        
        cin>>a>>b;
        head[a].push_back(b);
        head[b].push_back(a);
    }
    bfs(1);
    cin>>m;
    for(int i=1;i<=m;i++){
        cin>>a>>b;
        cout<<lca(a,b)<<endl;
    }
    return 0;
}

using namespace std;
/*
    tarjan 算法
    对暴力向上枚举的lca的一个并查集优化
    是一个离线查询
    所谓离线就是预处理了所有的答案，之后就是0（1）的输出答案即可
    牛 哇～～～
    
    节点类型： 
    0. 还未搜索的点
    1. 正在搜索的分支
    2. 已经遍历，且回溯了的点

	其中 dis[i] 数组表示i到root的距离，
	该数组可以通过dfs求得
	
	下面的(1)代表上面的节点类型	
    对于每个 （1）
    找到关于（1）的任务的（2）
    他们之间的最短距离为
    
    distance = dis[(1)] + dis[(2)] - 2 * dis[find((2))];
    
	关于 - 2 * dis[find((2))]通过画图很好理解
	就是dis[(2)]已经遍历并且回溯了，他就会通过并查集归类到他的父节点上
    因为dis[(1)] 和 dis[(2)] 算了两次 root 到 find((2)) 的距离
    那么对于求出(1) 到 (2)的距离，root 到 find（(2))的距离不需要算进去
	

*/
const int N = 1e4 + 10 , M = N * 2;
int p[N];
int n,m;
int e[M],h[N],ne[M],w[M],idx;
int dis[N];
int st[N];
int ans[M];
vector<pair<int,int >>q[N];

void add(int a,int b,int c){
    e[idx]=b,w[idx]=c,ne[idx]=h[a],h[a]=idx++;
}
// 计算dis[]数组
void dfs(int u,int fa){
    for(int i=h[u];i!=-1;i=ne[i]){
        int j=e[i];
        if(j == fa) continue;
        dis[j] = dis[u] + w[i];
        dfs(j,u);
    }
}
// 并查集
int find(int x){
    if(p[x]!=x){
        p[x]=find(p[x]);
    }
    return p[x];
}
// tarjan 
void tarjan(int u){

    st[u]=1;
    for(int i=h[u];i!=-1;i=ne[i]){
        int j=e[i];
        if(st[j]==0){
            trajan(j);
            p[j]=u;
        }
    }
    // 对于(1)的所有任务找到(2)
    for(auto x: q[u]){
        int vur = x.first;
        int id = x.second;
        if(st[vur]==2){
            ans[id]=dis[vur] + dis[u] - 2 * dis[find(vur)];
        }
    }
    st[u]=2;
}
int main()
{

    cin >> n >> m;
    int a,b,c;
    memset(h,-1,sizeof h);
    for(int i=1;i<=n-1;i++){
        cin>>a>>b>>c;
        add(a,b,c);
        add(b,a,c);
    }
    for(int i=1;i<=n;i++){
        p[i]=i;
    }
    for(int i=1;i<=m;i++){
        cin>>a>>b;
        q[a].push_back({b,i});
        q[b].push_back({a,i});
    }   
    dfs(1,-1);
    tarjan(1);
    for(int i=1;i<=m;i++){
        cout<<ans[i]<<endl;
    }

    return 0;
}

/*

	https://www.acwing.com/activity/content/problem/content/1540/
    LCA + 树上差分
    代码很好写，思路很难想，
    这也是我第一次遇见树上差分，还不知道如何想到他
    类比数组差分
    区间操作 -》 路径操作
    前缀和 -》 子树和

    树上差分:
    x -> y 权重为c
    差分: d[x] += c, d[y] += c, d[LCA(x,y)] -= 2c
    得到的效果就是 x 到 y 这一路径全部 +c 
    当然要达到这个效果还需要通过dfs
    通过dfs求出 F[x]：表示以x为根的子树各个节点的权值之和
    那么F[x] 其实就是 x 与它的父节点之间的边被覆盖的次数
    同样画图很好理解



*/
const int N = 1e5+100;
vector<int>head[N];
int dep[N];
int f[N][17];
int d[N];
int n,m;
// lca初始化
void bfs(){
    memset(dep,0x3f,sizeof(dep));
    dep[0]=0;
    dep[1]=1;

    queue<int>q;
    q.push(1);

    while(q.size()){
        int t=q.front();
        q.pop();

        for(auto x: head[t]){
            if(dep[x] > dep[t] + 1){
                dep[x]=dep[t]+1;
                q.push(x);
                f[x][0]=t;
                for(int k=1;k<=16;k++){
                    f[x][k]=f[f[x][k-1]][k-1];   
                }
            }
        }
    }
}
ll ans = 0;

// 树上差分
// 在这里并没有使用F[x]，而是直接通过dfs回溯算出 s
int dfs(int u,int fa){
    int res = d[u];
    for(auto x : head[u]){
        if(x==fa)continue;
        int s = dfs(x,u);   
        if(s == 1) ans+=1;
        else if(s == 0)ans+=m;
        res += s;
    }
    return res;
}

// lca
int lca(int a,int b){
    if(dep[a] < dep[b]) swap(a,b);
    for(int k=16;k>=0;k--){ // 到0
        if(dep[f[a][k]] >= dep[b]){
            a=f[a][k];
        }
    }

    if(a==b)return a;

    for(int k=16;k>=0;k--){ // 到0
        if(f[a][k] != f[b][k]){
            a=f[a][k];
            b=f[b][k];
        }
    }
    return f[a][0];
} 
int main()
{

    cin>>n>>m;
    int a,b;
    for(int i=1;i<=n-1;i++){
        cin>>a>>b;
        head[a].push_back(b);
        head[b].push_back(a);
    }
    bfs();
    for(int i=1;i<=m;i++){
        cin>>a>>b;
        int p =lca(a,b);
        d[a]++,d[b]++,d[p]-=2;
    }

    dfs(1,-1);
    cout<<ans<<endl;


    return 0;
}

bool st[N];
int n ,m,k;
int a,b;
int ans = 0;
int h[N],e[N*2],ne[N*2],idx,w[N*2];
void add(int a,int b,int c){
    e[idx] = b, w[idx]=c ,ne[idx]=h[a] ,h[a]=idx++;
}

// dp求树的直径
// 真的只能求树的直径长度
// dis[u] : 以u作为根节点出发走向以 u 为根的子树，可以到达的最远的节点的距离
// u 的子节点为 y1,y2,y3··
// dis[u] = max(dis[y1] + edge(u,y1) , dis[y2] + edge(u,y2)···)
// F[x] ： 经过节点x的最长链的长度
// F[x] = max(dis[y1] + dis[y2] + edge(x,y1) + edge(x,y2))

void dp(int u){
    st[u]=1;
    for(int i=h[u];i!=-1;i=ne[i]){
        int j = e[i];
        if(st[j]) continue;
        dp(j);
        //dis[u] 是以u为根的子远节点距离:思考递归的过程
        //dis[j] 是u的一个分支的树的最以j为根的子树的最远节点距离
        //w[i] = edge(u,j)
        ans = max(ans ,dis[u] + dis[j] + w[i] );
        dis[u] = max(dis[u], dis[j] + w[i] );
    } 
}



// dfs 求直径的两端
// 只适用于非负权变
// 1.对于任意一个节点找到他最远的节点  a
// 2.对于a 找到他最远的节点 b 
// 直径 = ab 
int id;
void dfs(int u,int fa,int sum){
    if(sum > ans){
        ans = sum;
        id = u;
    }
    for(int i = h[u]; i != -1; i = ne[i]){
        int j= e[i];
        if(j == fa ) continue;
        dfs(j, u ,sum + w[i]);
    }
}

//用邻接表
void dijk()
{
    priority_queue<PII, vector<PII>, greater<PII>> q;
    /*
    pair默认排序是按照
a.first==b.first?a.first<b.first:a.second>b.second;
    所以每次找最近的点,应该要把距离放在first
    */
    q.push({ 0,1});
    memset(dis, 0x3f, sizeof(dis));
    dis[1] = 0;
    while (q.size()) {
        auto t = q.top();
        q.pop();
        int vur = t.second  , distance = t.first;
        if (st[vur])
            continue;
        for (int i = h[vur]; i != -1; i = ne[i]) {
            int j = e[i];
            if (dis[j] > distance + w[i]) {
                dis[j] = distance + w[i];
                q.push({ dis[j],j});
            }
        }
        st[vur] = true;
    }
    if (dis[n] == 0x3f3f3f3f) {
        cout << -1 << endl;
        return;
    }
    cout << dis[n] << endl;
}

#include <bits/stdc++.h>

using namespace std;
/*  st表
    打表处理，求区间最大值
    f[i][j]: 表示[i,j]的最大值
    转移方程
        f[i][j] = a[j] (i==j)
        f[i][j] = max(f[i][j-1],a[j])
    时间：O(n^2)


    倍增
    f[i][j]:表示从i开始长度为2^j的区间的max，即[i,i + 2^j -1]的max
    f[i][j] = max(f[i][j-1],f[i+(1<<(j-1))][j-1]);
    
    时间复杂度：O(n)
*/

const int N = 10010;
int f[N][N];
int n = 7;
int a[] = { 9, 3, 1, 7, 5, 6, 0 };
// dp区间查询max
void common(){
    
    for (int i = 0; i < n; i++) {
        for (int j = i; j < n; j++) {
            if (i == j)
                f[i][j] = a[i];
            else
                f[i][j] = max(f[i][j - 1], a[j]);
        }
    }
}

// dp + 倍增法区间查询max
void mult(){
   // 需要先更新k-1的状态
   // 下标 0 ~ n-1
   // f[i][j] ：区间[i,i + 2^j -1 ]区间的max
    for(int i=0;i<n;i++)f[i][0]=a[i];
    for(int k = 1; k <= log2(n); k++){
        for(int i=0;i + (1<<k) -1 <n;i++){
            f[i][k] = max(f[i][k-1],f[i+(1<<(k-1))][k-1]);
        }
    }
}
void query(int l,int r){
    int len = r - l + 1;
    int j = log2(len);
   
	return max(f[l][j],f[r - (1<<j)+1][j]);
}
int main()
{

判断是否是4的幂次

如果 n是 44 的幂，那么 n 一定也是 2 的幂。因此我们可以首先判断 n 是否是 2 的幂，在此基础上再判断 n 是否是 4 的幂。

如果 n 是 4的幂，那么 n的二进制表示中有且仅有一个 1，并且这个 1 出现在从低位开始的第偶数个二进制位上（这是因为这个 1 后面必须有偶数个 0）。这里我们规定最低位为第 0 位，例如 n=16 时，n 的二进制表示为 (10000)

唯一的1出现在第4个二进制位上，因此n是4的幂

由于题目保证了 n是一个 32位的有符号整数，因此我们可以构造一个整数，它的所有偶数二进制位都是 0，所有奇数二进制位都是 1。这样一来，我们将 n和 num 进行按位与运算，如果结果为 0，说明 n 二进制表示中的 1 出现在偶数的位置，否则说明其出现在奇数的位置。

        {
            st[pri[j] * i] = true;
            if (i % pri[j] == 0) break;
        }
    }
}

小技巧

减少unordered_map时间

其实我们要从底层原理来了解为什么unordered_map会被卡，是因为它的load_factor超过
了设定的阈值max_load_factor就会引发哈希表的重建，这个时间耗费才是关键

我们姑且把load_factor称为负载因子，它等于无序桶的数目/所有元素的数目，个人经验max_load_factor值取0.25并将初始化桶数设置为1024可以最大化避免大数据hash的时间
(ps:这个最优化不一定是要求重建次数最小，而是在重建时这个表的规模大小和重建次数中做个平衡~)

摘抄

unordered_map<int ,int >mp;

mp.reserve(1024);
mp.max_load_factor(0.25);
